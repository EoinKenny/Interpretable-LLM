class ProtoSAE(nn.Module):

    def __init__(self, n_latents: int, n_inputs: int, prototype_dim: int) -> None:
        super().__init__()
        self.bias = nn.Parameter(torch.randn(n_latents))  # Initialize bias to -1

        self.encoders = nn.ModuleList(
            [self._build_mlp(n_inputs, prototype_dim, n_inputs // 4) for _ in range(n_latents*2)]
        )

        self.prototype_dim = prototype_dim
        self.n_latents = n_latents
        self.prototypes = nn.Parameter(torch.zeros(n_latents * 2, prototype_dim))  # Two prototypes per latent
        self.weights = nn.Parameter(torch.cat((torch.ones(n_latents), -torch.ones(n_latents))))  # Positive and negative weights
        self.epsilon = 1e-4  # A small value to avoid division by zero

    def _build_mlp(self, input_dim: int, output_dim: int, hidden_dim: int) -> nn.Sequential:
        return nn.Sequential(
            nn.Linear(input_dim, hidden_dim),  # First layer
            nn.BatchNorm1d(hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, output_dim)  # Second layer
        )
    
    def _proto_layer(self, x):
        """
        Inputs:
        x: a tensor of shape (batch_size, seq_len, n_latents, prototype_dim)
    
        The method calculates the L2 distance between each encoder's output and its corresponding prototypes,
        then computes the activation value for both prototypes and combines them using their respective weights.
        """
    
        b_size, seq_len, n_latents, prototype_dim = x.shape
    
        # Reshape prototypes to align properly for distance calculation
        c = self.prototypes.unsqueeze(0)
        x = x.view(b_size*seq_len, n_latents, prototype_dim)
        
        # Calculate L2 distances
        l2s = ((x - c) ** 2).sum(dim=-1)
    
        # Calculate activations
        act = torch.log((l2s + 1.) / (l2s + self.epsilon)) 
        act *= proto_sae.weights

        # Split the tensor into the first and second groups of 10 elements
        first_group = act[:, :10]  # Shape: (3048, 10)
        second_group = act[:, 10:]  # Shape: (3048, 10)
        
        # Sum the two groups element-wise
        act_summed = first_group + second_group  # Shape: (3048, 10)
            
        return act_summed

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Inputs:
        x is tensor of shape (batch_size, sequence_len, n_inputs)
        """        
        # Encode the input using all encoders
        encoded = [encoder(x.view(-1, x.shape[-1])) for encoder in self.encoders]
        encoded = torch.stack(encoded, dim=1)  
        encoded = encoded.view(x.shape[0], x.shape[1], self.n_latents*2, self.prototype_dim)

        # Pass through the prototype layer
        proto_activations = self._proto_layer(encoded) 

        # # Add the bias, multiply by the scaling factor alpha, and apply ReLU
        # latents = proto_activations + self.bias  # Add bias term

        return proto_activations.view(x.shape[0], x.shape[1], self.n_latents)

    def constrain_weights(self):
        """Constrain the weights to maintain their polarity during training."""
        with torch.no_grad():
            self.weights[:self.weights.size(0) // 2].clamp_(min=0)  # Ensure positive weights remain positive
            self.weights[self.weights.size(0) // 2:].clamp_(max=0)  # Ensure negative weights remain negative
